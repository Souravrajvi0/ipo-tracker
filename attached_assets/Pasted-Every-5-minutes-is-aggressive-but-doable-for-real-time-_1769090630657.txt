function every5Minutes() {
  const now = new Date();
  
  // Only poll during bidding hours (IST)
  if (isBiddingHours(now)) {
    const chittorgarhData = scrapeChittorgarh();
    const nseData = scrapeNSE();
    
    const delta = compare(chittorgarhData, previousSnapshot);
    
    if (delta.subscriptionChange > 5) {  // 5% jump
      alertTelegram(`‚ö†Ô∏è ${symbol} subscription jumped: ${delta}%`);
    }
    
    store(delta);
    updatePreviousSnapshot(chittorgarhData);
  }
  
  // Always track post-listing prices
  if (isListed) {
    const cmp = fetchNSEPrice(symbol);
    const pnl = calculatePnL(cmp, buyPrice);
    
    if (pnl < -10) {  // Stop loss
      alertTelegram(`üî¥ ${symbol} P&L: ${pnl}% - STOP LOSS`);
    }
  }
}
